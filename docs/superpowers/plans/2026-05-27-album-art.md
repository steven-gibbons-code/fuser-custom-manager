# Album Art Display Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Resolve real album art from FSL API, MusicBrainz CAA, and Google Drive; cache images to disk; display them in the song list rows and detail panel.

**Architecture:** Art URLs are resolved at sync time and stored in a new `art_url` DB column. FSL songs get their URL from the `poster` JSON field. Fucuco songs are looked up via MusicBrainz CAA then a scraped Google Drive folder index. A background `ArtFetchWorker` downloads images to `~/.fuser_manager/art/{song_id}.jpg` after sync. `_art_pixmap` checks the disk cache first before generating a gradient.

**Tech Stack:** PySide6 6.x, SQLite via `sqlite3`, `requests`, MusicBrainz WS2 API (no key), Google Drive public folder scraping (no key, same mechanism as gdown)

```python
import time
import requests

_MB_URL = "https://musicbrainz.org/ws/2/release"
_CAA_URL = "https://coverartarchive.org/release/{mbid}/front-250"
_USER_AGENT = "FuserCustomTool/1.0 (sgibb.code@gmail.com)"

_last_call = 0.0


def _throttle():
    global _last_call
    elapsed = time.time() - _last_call
    if elapsed < 1.0:
        time.sleep(1.0 - elapsed)
    _last_call = time.time()


def musicbrainz_lookup(artist: str, title: str) -> str | None:
    """Return a Cover Art Archive image URL for the given artist+title, or None."""
    try:
        _throttle()
        resp = requests.get(
            _MB_URL,
            params={"query": f'artist:"{artist}" recording:"{title}"', "fmt": "json", "limit": 5},
            headers={"User-Agent": _USER_AGENT},
            timeout=10,
        )
        if resp.status_code != 200:
            return None
        releases = resp.json().get("releases", [])
        if not releases:
            return None
        mbid = releases[0]["id"]

        _throttle()
        caa_resp = requests.get(
            _CAA_URL.format(mbid=mbid),
            headers={"User-Agent": _USER_AGENT},
            timeout=10,
            allow_redirects=True,
        )
        if caa_resp.status_code == 200:
            return caa_resp.url
        return None
    except Exception:
        return None
```

```python
import json
import re
import time
from pathlib import Path

import requests

FOLDER_ID = "14r8__8RAlxPc278yAe82ukq-aiRpgpC0"
_INDEX_PATH = Path.home() / ".fuser_manager" / "gdrive_art_index.json"

def _parse_html(html: str) -> list[dict]:
    """
    Extract [{id, name, children: [{id, name}]}] from a GDrive folder page.

    Google Drive embeds folder data inside AF_initDataCallback calls.
    The format is an array-of-arrays where each entry is:
      [file_id, file_name, null, null, null, [children_if_folder], ...]
    and image files have a MIME type string in a later position.

    IMPORTANT: Adapt the regex below based on the actual HTML you captured in
    Step 1 of this task. The pattern may differ from this approximation.
    """
    # Extract all AF_initDataCallback payloads
    payloads = re.findall(
        r'AF_initDataCallback\(\{[^}]*"data"\s*:\s*(\[.*?\])\s*\}\)',
        html,
        re.DOTALL,
    )
    entries = []
    for payload in payloads:
        try:
            data = json.loads(payload)
            # Walk nested arrays to find file/folder entries
            # An entry looks like: ["ID", "Name", null, ...]
            # A folder entry has a list at index 5: ["ID", "Name", null, null, null, [...]]
            _walk(data, entries)
        except (json.JSONDecodeError, TypeError, IndexError):
            continue
    return entries


def _walk(node, entries: list):
    """Recursively find [id, name, ...] entries that look like Drive file/folder records."""
    if not isinstance(node, list):
        return
    # A Drive entry: first element is a string ID (~25-33 chars), second is a string name
    if (len(node) >= 2
            and isinstance(node[0], str) and len(node[0]) >= 20
            and isinstance(node[1], str) and node[1]):
        children = []
        # If element at index 5 is a list, this is a folder with children
        if len(node) > 5 and isinstance(node[5], list):
            for child in node[5]:
                if (isinstance(child, list) and len(child) >= 2
                        and isinstance(child[0], str) and isinstance(child[1], str)):
                    children.append({"id": child[0], "name": child[1]})
        entries.append({"id": node[0], "name": node[1], "children": children})
        return
    for item in node:
        _walk(item, entries)


def _filter_images(files: list[dict]) -> list[dict]:
    img_exts = {".jpg", ".jpeg", ".png", ".webp"}
    return [f for f in files if Path(f["name"]).suffix.lower() in img_exts]


def build_index(folder_id: str, html: str | None = None) -> dict:
    """
    Return {artist_name_lower: {files: [{id, name}]}} for a public GDrive folder.

    Pass `html` directly in tests. In production, html=None triggers a real HTTP fetch.
    """
    if html is None:
        resp = requests.get(
            f"https://drive.google.com/drive/folders/{folder_id}",
            headers={"User-Agent": "Mozilla/5.0"},
            timeout=30,
        )
        resp.raise_for_status()
        html = resp.text

    entries = _parse_html(html)
    index = {}
    for entry in entries:
        artist_key = entry["name"].strip().lower()
        images = _filter_images(entry["children"])
        if images:
            index[artist_key] = {"files": images}
    return index


def _load_index() -> dict:
    """Load index from disk cache if fresh; otherwise build from Drive."""
    if _INDEX_PATH.exists():
        try:
            data = json.loads(_INDEX_PATH.read_text(encoding="utf-8"))
            if time.time() - data.get("ts", 0) < _TTL:
                return data["index"]
        except (json.JSONDecodeError, KeyError):
            pass

    index = build_index(FOLDER_ID)
    _INDEX_PATH.parent.mkdir(parents=True, exist_ok=True)
    _INDEX_PATH.write_text(
        json.dumps({"ts": time.time(), "index": index}),
        encoding="utf-8",
    )
    return index


def get_index() -> dict:
    return _load_index()


def lookup(artist: str) -> str | None:
    """Return a direct download URL for the first image matching the artist, or None."""
    key = artist.strip().lower()
    index = _load_index()
    entry = index.get(key)
    if not entry or not entry.get("files"):
        return None
    file_id = entry["files"][0]["id"]
    return f"https://drive.google.com/uc?id={file_id}&export=download"
```

class ArtFetchWorker(QThread):
    art_ready = Signal(int)
    finished = Signal()

    def __init__(self, songs: list[dict], parent=None):
        super().__init__(parent)
        self._songs = songs  # each dict has {id, art_url}

    def run(self):
        ART_DIR.mkdir(parents=True, exist_ok=True)
        for song in self._songs:
            song_id = song["id"]
            dest = ART_DIR / f"{song_id}.jpg"
            if dest.exists():
                continue
            try:
                resp = requests.get(
                    song["art_url"],
                    timeout=15,
                    headers={"User-Agent": "Mozilla/5.0"},
                )
                if resp.status_code == 200:
                    dest.write_bytes(resp.content)
                    self.art_ready.emit(song_id)
            except Exception:
                pass
        self.finished.emit()
```

def _art_pixmap(song_id: int, size: int = ART_SIZE) -> QPixmap:
    """Return a cached pixmap for song_id: real art from disk, or gradient fallback."""
    key = f"art_{song_id}_{size}"
    pm = QPixmap()
    if QPixmapCache.find(key, pm):
        return pm

    # Check disk cache for downloaded art
    art_file = ART_DIR / f"{song_id}.jpg"
    if art_file.exists():
        source = QPixmap(str(art_file))
        if not source.isNull():
            # Scale to fill, then crop to exact square
            scaled = source.scaled(
                size, size,
                Qt.AspectRatioMode.KeepAspectRatioByExpanding,
                Qt.TransformationMode.SmoothTransformation,
            )
            x = (scaled.width() - size) // 2
            y = (scaled.height() - size) // 2
            cropped = scaled.copy(x, y, size, size)

            # Apply rounded corners via clip path
            rounded = QPixmap(size, size)
            rounded.fill(Qt.GlobalColor.transparent)
            p = QPainter(rounded)
            p.setRenderHint(QPainter.RenderHint.Antialiasing)
            clip = QPainterPath()
            clip.addRoundedRect(0, 0, size, size, ART_RADIUS, ART_RADIUS)
            p.setClipPath(clip)
            p.drawPixmap(0, 0, cropped)
            p.end()
            QPixmapCache.insert(key, rounded)
            return rounded

    # Generate gradient placeholder
    pm = QPixmap(size, size)
    pm.fill(Qt.GlobalColor.transparent)
    p = QPainter(pm)
    p.setRenderHint(QPainter.RenderHint.Antialiasing)
    a, b = _PALETTES[abs(song_id) % len(_PALETTES)]
    g = QLinearGradient(0, 0, size, size)
    g.setColorAt(0, QColor(a))
    g.setColorAt(1, QColor(b))
    path = QPainterPath()
    path.addRoundedRect(0, 0, size, size, ART_RADIUS, ART_RADIUS)
    p.fillPath(path, QBrush(g))
    p.setPen(QPen(QColor(255, 255, 255, 20), 1))
    p.setBrush(Qt.BrushStyle.NoBrush)
    p.drawPath(path)
    p.end()
    QPixmapCache.insert(key, pm)
    return pm
```
